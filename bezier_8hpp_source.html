269  const double dt = *(it_t + 1) - *it_t;
270 
271  // scaled velocity

273

276

278  Eigen::Matrix<Scalar, Impl::RepSize, 1> tmp1, tmp2;
279

281  Impl::composition(g_map, tmp1, tmp2); // tmp2 holds g * exp(va / 2)
282

284

286

291  }

313  throw std::runtime_error("Not enough points");
314  }

316  using G = std::ranges::range_value_t<Rg>;
317  using Scalar = typename G::Scalar;
318  using V = Eigen::Matrix<typename G::Scalar, lie_traits<G>::Impl::Dof, 1>;
319  using Impl = typename lie_traits<G>::Impl;
320 
321  // number of intervals
322  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
323 
324  std::size_t NumVars = Impl::Dof * 3 * N;
325 
326  Eigen::SparseMatrix<Scalar> lhs;
327  lhs.resize(NumVars, NumVars);
328  Eigen::Matrix<int, -1, 1> nnz = Eigen::Matrix<int, -1, 1>::Constant(NumVars, 3);
329  nnz.head(Impl::Dof).setConstant(2);
330  nnz.tail(Impl::Dof).setConstant(2);
331  lhs.reserve(nnz);
332 
333  Eigen::Matrix<Scalar, -1, 1> rhs(NumVars);
334  rhs.setZero();
335 
336  // variable layout:
337  //
338  // [ v_{1, 0}; v_{2, 0}; v_{3, 0}; v_{1, 1}; v_{2, 1}; v_{3, 1}; ...]
339  //
340  // where v_ji is a Dof-length vector
341 
342  const auto idx = [&](int j, int i) { return 3 * Impl::Dof * i + Impl::Dof * (j - 1); };
343 
344  std::size_t row_counter = 0;
345 
347 
348  // zero second derivative at start:
349  // v_{1, 0} = v_{2, 0}
350  const std::size_t v10_start = idx(1, 0);
351  const std::size_t v20_start = idx(2, 0);
352  for (auto n = 0u; n != Impl::Dof; ++n) {
353  lhs.insert(row_counter + n, v10_start + n) = 1;
354  lhs.insert(row_counter + n, v20_start + n) = -1;
355  }
356  row_counter += Impl::Dof;
357 
359 
360  auto it_t = std::ranges::begin(tt);
361  auto it_g = std::ranges::begin(gg);
362 
363  for (auto i = 0u; i + 1 < N; ++i, ++it_t, ++it_g) {
364  const std::size_t v1i_start = idx(1, i);
365  const std::size_t v2i_start = idx(2, i);
366  const std::size_t v3i_start = idx(3, i);
367 
368  const std::size_t v1ip_start = idx(1, i + 1);
369  const std::size_t v2ip_start = idx(2, i + 1);
370 
371  // segment lengths
372  const Scalar Ti = *(it_t + 1) - *it_t;
373  const Scalar Tip = *(it_t + 2) - *(it_t + 1);
374 
375  // pass through control points
376  // v_{1, i} + v_{2, i} + v_{3, i} = x_{i+1} - x_i
377  for (auto n = 0u; n != Impl::Dof; ++n) {
378  lhs.insert(row_counter + n, v1i_start + n) = 1;
379  lhs.insert(row_counter + n, v2i_start + n) = 1;
380  lhs.insert(row_counter + n, v3i_start + n) = 1;
381  }
382  rhs.segment(row_counter, Impl::Dof) = *(it_g + 1) - *(it_g);
383  row_counter += Impl::Dof;
384 
385  // velocity continuity
386  // v_{3, i} = v_{1, i+1}
387  for (auto n = 0u; n != Impl::Dof; ++n) {
388  lhs.insert(row_counter + n, v3i_start + n) = 1 * Tip;
389  lhs.insert(row_counter + n, v1ip_start + n) = -1 * Ti;
390  }
391  row_counter += Impl::Dof;
392 
393  // acceleration continuity (approximate for Lie groups)
394  // v_{2, i} - v_{3, i} = v_{2, i+1} - v_{1, i+1}
395  for (auto n = 0u; n != Impl::Dof; ++n) {
396  lhs.insert(row_counter + n, v2i_start + n) = 1 * (Tip * Tip);
397  lhs.insert(row_counter + n, v3i_start + n) = -1 * (Tip * Tip);
398  lhs.insert(row_counter + n, v1ip_start + n) = -1 * (Ti * Ti);
399  lhs.insert(row_counter + n, v2ip_start + n) = 1 * (Ti * Ti);
400  }
401  row_counter += Impl::Dof;
402  }
403 
405 
406  const std::size_t v1_nm_start = idx(1, N - 1);
407  const std::size_t v2_nm_start = idx(2, N - 1);
408  const std::size_t v3_nm_start = idx(3, N - 1);
409 
410  // end at last control point
411  // v_{1, n-1} + v_{2, n-1} v_{3, n-1} = x_{n} - x_{n-1}
412  for (auto n = 0u; n != Impl::Dof; ++n) {
413  lhs.insert(row_counter + n, v1_nm_start + n) = 1;
414  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
415  lhs.insert(row_counter + n, v3_nm_start + n) = 1;
416  }
417  rhs.segment(row_counter, Impl::Dof) = *(it_g + 1) - *it_g;
418  row_counter += Impl::Dof;
419 
420  // zero second derivative at end:
421  // v_{2, n-1} = v_{3, n-1}
422  for (auto n = 0u; n != Impl::Dof; ++n) {
423  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
424  lhs.insert(row_counter + n, v3_nm_start + n) = -1;
425  }

430

434  Eigen::VectorXd result = solver.solve(rhs);
435 
437 
438  std::vector<Bezier<3, G>> segments;
439  segments.reserve(N);
440 
441  it_g = std::ranges::begin(gg);
442 
443  for (auto i = 0u; i != N; ++i, ++it_g) {
444  const std::size_t v1i_start = idx(1, i);
445  const std::size_t v3i_start = idx(3, i);
446 
447  V v1 = result.template segment<Impl::Dof>(v1i_start);
448  V v3 = result.template segment<Impl::Dof>(v3i_start);
449  // re-compute v2 to compensate for linearization
450  // this ensures points are interpolated, but the cost is
451  // potential non-continuity of the second derivative
452  Eigen::Map<const Eigen::Matrix<Scalar, Impl::RepSize, 1>> g_map(it_g->data());
453  Eigen::Map<const Eigen::Matrix<Scalar, Impl::RepSize, 1>> gp_map((it_g + 1)->data());
454 
455  Eigen::Matrix<Scalar, Impl::RepSize, 1> tmp1, tmp2, tmp3;
456 
457  Impl::exp(v1, tmp1); // tmp1 holds exp(v1)
458  Impl::composition(g_map, tmp1, tmp2); // tmp2 holds g * exp(v1)
459  Impl::exp(-v3, tmp1); // tmp1 holds exp(-v3)
460  Impl::composition(gp_map, tmp1, tmp3); // tmp3 holds gp * exp(-v3)
461 
462  Eigen::Map<const G> g_tmp2(tmp2.data());
463  Eigen::Map<const G> g_tmp3(tmp3.data());
464 
465  V v2 = g_tmp3 - g_tmp2;
466 
467  segments.emplace_back(*it_g, std::array<V, 3>{std::move(v1), std::move(v2), std::move(v3)});
468  }
