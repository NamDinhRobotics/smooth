69 template<typename _F, typename _Wrt>
70 auto dr_numerical(_F && f, _Wrt && x)
71 {
72  using Result = typename decltype(std::apply(f, x))::PlainObject;
73  using Scalar = typename Result::Scalar;
74 
75  // static sizes
76  static constexpr Eigen::Index Nx = utils::tuple_dof<std::decay_t<_Wrt>>::value;
77  static constexpr Eigen::Index Ny = Result::SizeAtCompileTime;
78 
79  const Scalar eps = std::sqrt(Eigen::NumTraits<Scalar>::epsilon());
80 
81  const Result val = std::apply(f, x);
82 
83  // dynamic sizes
84  Eigen::Index nx = std::apply([](auto &&... args) { return (args.size() + ...); }, x);
85  Eigen::Index ny = val.size();
86 
87  // output variable
88  Eigen::Matrix<Scalar, Ny, Nx> jac(ny, nx);
89 
90  Eigen::Index index_pos = 0;
91 
92  utils::static_for<std::tuple_size_v<std::decay_t<_Wrt>>>([&](auto i) {
93  static constexpr Eigen::Index Nx_j =
94  std::decay_t<std::tuple_element_t<i, std::decay_t<_Wrt>>>::SizeAtCompileTime;
95  auto & w = std::get<i>(x);
96  const int nx_j = w.size();
97 
98  using W = std::decay_t<decltype(w)>;
99 
100  for (auto j = 0; j != nx_j; ++j) {
101  Scalar eps_j = eps;
102  if constexpr (std::is_base_of_v<Eigen::MatrixBase<W>, W>) {
103  // scale step size if we are in Rn
104  eps_j *= abs(w[j]);
105  if (eps_j == 0.) { eps_j = eps; }
106  } else if constexpr (std::is_base_of_v<smooth::TnBase<W>, W>) {
107  // or Tn
108  eps_j *= abs(w.rn()[j]);
109  if (eps_j == 0.) { eps_j = eps; }
110  }
111  // const cast needed in case argument is const (value is restored two lines below)
112  const_cast<W &>(w) += (eps_j * Eigen::Matrix<Scalar, Nx_j, 1>::Unit(nx_j, j));
113  jac.col(index_pos + j) = (std::apply(f, x) - val) / eps_j;
114  const_cast<W &>(w) += (-eps_j * Eigen::Matrix<Scalar, Nx_j, 1>::Unit(nx_j, j));
115  }
116  index_pos += nx_j;
117  });
118 
119  return std::make_pair(val, jac);
120 }

160 template<Type dm, typename _F, typename _Wrt>
161 auto dr(_F && f, _Wrt && x)
162 {
163  if constexpr (dm == Type::NUMERICAL) {
164  return detail::dr_numerical(std::forward<_F>(f), std::forward<_Wrt>(x));
165  } else if constexpr (dm == Type::AUTODIFF) {
166 #ifdef SMOOTH_DIFF_AUTODIFF
167  return dr_autodiff(std::forward<_F>(f), std::forward<_Wrt>(x));
168 #else
169  static_assert(dm != Type::AUTODIFF, "compat/autodiff.hpp header not included");
170 #endif
171  } else if constexpr (dm == Type::CERES) {
172 #ifdef SMOOTH_DIFF_CERES
173  return dr_ceres(std::forward<_F>(f), std::forward<_Wrt>(x));
174 #else
175  static_assert(dm != Type::CERES, "compat/ceres.hpp header not included");
176 #endif
177  } else if constexpr (dm == Type::ANALYTIC) {
178  return std::apply(f, std::forward<_Wrt>(x));
179  } else if constexpr (dm == Type::DEFAULT) {
180  return dr<DefaultType>(std::forward<_F>(f), std::forward<_Wrt>(x));
181  }
182 }
