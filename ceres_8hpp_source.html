87 template<typename _F, typename _Wrt>
88 auto dr_ceres(_F && f, _Wrt && x)
89 {
90  using Result = decltype(std::apply(f, x));
91  using Scalar = typename Result::Scalar;
92 
93  const auto fval = std::apply(f, x);
94 
95  static constexpr Eigen::Index Nx = utils::tuple_dof<_Wrt>::value;
96  const auto nx = std::apply([](auto &&... args) { return (args.size() + ...); }, x);
97  static constexpr Eigen::Index Ny = Result::SizeAtCompileTime;
98  const auto ny = fval.size();
99 
100  Eigen::Matrix<Scalar, Nx, 1> a(nx);
101  a.setZero();
102 
103  Eigen::Matrix<Scalar, Ny, 1> b(ny);
104 
105  Eigen::Matrix<Scalar, Ny, Nx, (Nx == 1) ? Eigen::ColMajor : Eigen::RowMajor> jac(ny, nx);
106 
107  const Scalar * prms[1] = {a.data()};
108  double ** jac_rows = new double *[ny];
109  for (auto i = 0; i < ny; i++) { jac_rows[i] = jac.row(i).data(); }
110 
111  const auto f_deriv = [&]<typename T>(const T * in, T * out) {
112  Eigen::Map<const Eigen::Matrix<T, Nx, 1>> m_in(in, nx);
113  Eigen::Map<Eigen::Matrix<T, Ny, 1>> m_out(out, ny);
114 
115  m_out =
116  std::apply(f, utils::tuple_plus(utils::tuple_cast<T>(x), m_in)) - fval.template cast<T>();
117 
118  return true;
119  };
120 
121  ceres::internal::AutoDifferentiate<Result::SizeAtCompileTime,
122  ceres::internal::StaticParameterDims<Nx>>(f_deriv, prms, b.size(), b.data(), jac_rows);
123 
124  Eigen::Matrix<Scalar, Ny, Nx> jac_ret = jac;
125 
126  return std::make_pair(fval, jac_ret);
127 }
